Modelling of R with only Incidence data

This vignette describes how changes in (R_t) can be modelled with just incidence data and discrete SI.

Flu 1918

We use the Flu data from 1918 as an example.

library(epidemia)
data("Flu1918")
options(mc.cores = parallel::detectCores())
print(Flu1918)
## $incidence
##  [1]   5   1   6  15   2   3   8   7   2  15   4  17   4  10  31  11  13  36  13
## [20]  33  17  15  32  27  70  58  32  69  54  80 405 192 243 204 280 229 304 265
## [39] 196 372 158 222 141 172 553 148  95 144  85 143  87  73  70  62 116  44  38
## [58]  60  45  60  27  51  34  22  16  11  18  11  10   8  13   3   3   6   6  13
## [77]   5   6   6   5   5   1   2   2   3   8   4   1   2   3   1   0
## 
## $si_distr
##  [1] 0.000 0.233 0.359 0.198 0.103 0.053 0.027 0.014 0.007 0.003 0.002 0.001

flu <- Flu1918
flu$incidence <- c(rep(NA,1),flu$incidence) ## pad before initialisation
flu$fludate <- as.Date("1918-01-01")+seq(0,along.with=flu$incidence)
obs <- !is.na(flu$incidence)

args <- list(formula=Rt(country,date)~rw(date,3),
             data=data.frame(country="A",date=flu$fludate),
             obs=list(
                 incidence=list(
                     odata=data.frame(country="A",
                                      date=flu$fludate[obs],incidence=flu$incidence[obs]),
                     rates=list(means=data.frame(factor("A"),1),
                                scale=.01),
                     pvec=c(.25,.25,.25,.25)
                 )
             ),
             seed_days=7,
             algorithm="sampling",
             r0=3,
             pops=data.frame(country="A",pop=1e6),
             si=flu$si,
             prior = rstanarm::normal(location=0,scale=.2),
             prior_intercept = rstanarm::normal(location=0,scale=.5),
             prior_tau = rstanarm::exponential(rate=4)             
             )
args$sampling_args <- list(iter=1000,control=list(adapt_delta=0.95,max_treedepth=15),seed=713)
fit <- xfun::cache_rds(do.call("epim",args),hash=args)
library(gridExtra)
grid.arrange(plot_rt(fit),
             plot_obs(fit,"incidence"),
             nrow=1)

SARS 2003 - cumulative cases

We use the SARS data from 2003 as an example. Fitting to cumulative cases. Currently independence between the cumulative counts is being assumed in the model - which probably should be relaxed.

library(epidemia)
data("SARS2003")
options(mc.cores = parallel::detectCores())
print(SARS2003)
## $incidence
##   [1]   1   0   0   1   0   0   2   0   2   2   1   1   1   0   0   0   4   1
##  [19]   2   4  13  23  35  26  12  17  19  17  28  23  27  27  11  21  21  25
##  [37]  31 103  96  69  58  48  33  25  43  37  30  29  28  34  34  32  24  17
##  [55]  16  23  18  15  11  19  16   9  17  14   6   4   8   9  11  11  11   6
##  [73]  13   3   8   2   4   4   6   4   5   4   8   2   3   2   5   3   3   4
##  [91]   3   1   2   0   1   3   3   1   1   1   1   1   1   1   0   2   0
## 
## $si_distr
##  [1] 0.000 0.001 0.012 0.043 0.078 0.104 0.117 0.116 0.108 0.094 0.078 0.063
## [13] 0.049 0.038 0.028 0.021 0.015 0.011 0.008 0.005 0.004 0.003 0.002 0.001
## [25] 0.001

sars <- SARS2003
sars$incidence <- c(rep(NA,20),cumsum(sars$incidence)) ## pad before initialisation
sars$sarsdate <- as.Date("2003-01-01")+seq(0,along.with=sars$incidence)
obs <- !is.na(sars$incidence)

args <- list(formula=Rt(country,date)~rw(date,3),
     data=data.frame(country="A",date=sars$sarsdate),
             obs=list(
                 incidence=list(
                     odata=data.frame(country="A",
                                      date=sars$sarsdate[obs],incidence=sars$incidence[obs]),
                     rates=list(means=data.frame(factor("A"),1),
                                scale=.01),
                     pvec=c(.25,.5,.75,1),
                     ptype="distribution"
                 )
             ),
             seed_days=7,
             algorithm="sampling",
             r0=3,
             pops=data.frame(country="A",pop=1e6),
             si=sars$si,
             prior = rstanarm::normal(location=0,scale=.2),
             prior_intercept = rstanarm::normal(location=0,scale=.5),
             prior_tau = rstanarm::exponential(rate=4)             
     )
args$sampling_args <- list(iter=1000,control=list(adapt_delta=0.95,max_treedepth=15),seed=713)
fit <- xfun::cache_rds(do.call("epim",args),hash=args)
library(gridExtra)
grid.arrange(plot_rt(fit),
             plot_obs(fit,"incidence"),
             plot_infections(fit),
             nrow=2)
